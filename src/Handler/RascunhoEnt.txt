getBuscarEntradaR :: EntradaId -> Handler Html
getBuscarEntradaR = error "undefined"

putEditarEntradaR :: EntradaId -> Handler Value
putEditarEntradaR = do
    _ <- runDB $ get404 EntradaId
    novaEntrada <- requireJsonBody :: Handler Entrada
    runDB $ replace entradaId novaEntrada
    sendStatusJSON noContent204 (object ["resp" .= ("UPDATED " ++ show (fromSqlKey entradaId))])

postApagarEntradaR :: EntradaId -> Handler Html
postApagarEntradaR eid = do 
    _ <- runDB $ get404 eid
    runDB $ delete eid 
    redirect ListarEntradaR
